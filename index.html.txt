<script>
let items=[{desc:'Website Design',qty:1,rate:300},{desc:'Logo Design',qty:1,rate:150}];
function addItem(){items.push({desc:'',qty:1,rate:0});renderItems();update()}
function removeItem(i){items.splice(i,1);renderItems();update()}
function renderItems(){
  const c=document.getElementById('itemsContainer');
  c.innerHTML=items.map((it,i)=>`<div class="item-row">
    <input value="${it.desc}" oninput="items[${i}].desc=this.value;update()" placeholder="Service...">
    <input type="number" value="${it.qty}" min="1" oninput="items[${i}].qty=+this.value;update()">
    <input type="number" value="${it.rate}" min="0" oninput="items[${i}].rate=+this.value;update()">
    <button class="del-btn" onclick="removeItem(${i})">×</button>
  </div>`).join('');
}
function fmt(n){return'$'+n.toFixed(2)}
function update(){
  const sub=items.reduce((s,it)=>s+(it.qty*it.rate),0);
  const tax=sub*(+(document.getElementById('taxRate').value||0)/100);
  const disc=sub*(+(document.getElementById('discRate').value||0)/100);
  const grand=sub+tax-disc;
  document.getElementById('subtotalDisp').textContent=fmt(sub);
  document.getElementById('taxDisp').textContent=fmt(tax);
  document.getElementById('discDisp').textContent='-'+fmt(disc);
  document.getElementById('grandDisp').textContent=fmt(grand);
  document.getElementById('p-invnum').textContent=document.getElementById('invNum').value||'INV-001';
  document.getElementById('p-bizname').textContent=document.getElementById('bizName').value;
  document.getElementById('p-bizdetail').innerHTML=document.getElementById('bizEmail').value+'<br>'+document.getElementById('bizPhone').value+'<br>'+document.getElementById('bizAddr').value.replace(/\n/g,'<br>');
  document.getElementById('p-client').textContent=document.getElementById('clientName').value;
  document.getElementById('p-clientemail').textContent=document.getElementById('clientEmail').value;
  const dd=document.getElementById('dueDate').value;
  document.getElementById('p-due').textContent=dd?new Date(dd).toLocaleDateString('en-PK',{day:'numeric',month:'short',year:'numeric'}):'—';
  document.getElementById('p-notes').textContent=document.getElementById('notes').value;
  document.getElementById('p-items').innerHTML=items.map(it=>`<tr><td>${it.desc||'—'}</td><td>${it.qty}</td><td>${fmt(it.rate)}</td><td style="text-align:right">${fmt(it.qty*it.rate)}</td></tr>`).join('');
  document.getElementById('p-sub').textContent=fmt(sub);
  document.getElementById('p-tax').textContent=fmt(tax);
  document.getElementById('p-disc').textContent='-'+fmt(disc);
  document.getElementById('p-grand').textContent=fmt(grand);
}
function clearAll(){document.querySelectorAll('input,textarea').forEach(el=>{if(el.type!=='number')el.value='';else el.value='0'});items=[{desc:'',qty:1,rate:0}];renderItems();update()}
function printInv(){
  const prev=document.getElementById('preview').innerHTML;
  const w=window.open('','_blank');
  w.document.write('<html><head><title>Invoice</title><style>body{font-family:sans-serif;padding:40px;max-width:700px;margin:auto}.inv-header{display:flex;justify-content:space-between;padding-bottom:16px;border-bottom:2px solid #185FA5;margin-bottom:20px}.inv-title{font-size:24px;font-weight:700;color:#185FA5}.inv-num,.biz-detail{font-size:12px;color:#888}.biz-name{font-size:15px;font-weight:600}.inv-meta{display:grid;grid-template-columns:1fr 1fr;gap:12px;margin-bottom:16px}.meta-box{background:#f5f5f0;border-radius:8px;padding:10px 12px}.meta-label{font-size:10px;color:#888;text-transform:uppercase}.meta-val{font-size:13px;font-weight:600}.inv-table{width:100%;border-collapse:collapse;font-size:12px;margin-bottom:16px}.inv-table th{background:#e6f1fb;color:#0c447c;padding:7px 8px;text-align:left}.inv-table td{padding:7px 8px;border-bottom:1px solid #eee}.inv-totals{margin-left:auto;width:200px}.inv-total-row{display:flex;justify-content:space-between;font-size:12px;padding:3px 0;color:#888}.inv-grand{display:flex;justify-content:space-between;font-size:14px;font-weight:700;padding:8px 0 0;color:#185FA5;border-top:2px solid #185FA5;margin-top:6px}.inv-footer{margin-top:16px;padding-top:12px;border-top:1px solid #eee;font-size:11px;color:#aaa}</style></head><body>'+prev+'</body></html>');
  w.document.close();w.print();
}
